set pagination off
set confirm off
target extended-remote :3333

echo \n=== REGISTERS ===\n
info registers

echo \n=== CURRENT TASK (currentTask) ===\n
# Try to read currentTask pointer - INAV's custom scheduler
print (void*)currentTask
x/2x (void*)&currentTask

echo \n=== BACKTRACE ===\n
bt
bt full

echo \n=== CURRENT FRAME ===\n
info frame

echo \n=== DISASSEMBLY AT PC ===\n
disassemble $pc-20,$pc+20

echo \n=== STACK TRACE ===\n
x/32x $sp

echo \n=== LR (Link Register - previous PC) ===\n
info registers lr

echo \n=== SCHEDULER TASK QUEUE ===\n
# Read task queue array - first entry is usually currentTask
print (void*)taskQueueArray
x/16x (void*)taskQueueArray

echo \n=== TASK TIMING ===\n
# Read scheduler statistics if available
print (void*)totalWaitingTasks
print (void*)averageSystemLoadPercent

quit
